CREATE TABLE DASHBOARDGRAFIK.V_SYSCODE
(
  ID            NUMBER(19),
  KODE          VARCHAR2(255 BYTE),
  VALUE_CHAR    VARCHAR2(255 BYTE),
  VALUE_NUMBER  VARCHAR2(255 BYTE),
  KET_CHAR      VARCHAR2(255 BYTE),
  KET_NUMBER    VARCHAR2(255 BYTE),
  CREATED_AT    TIMESTAMP(6),
  UPDATED_AT    TIMESTAMP(6)
)

CREATE SEQUENCE DASHBOARDGRAFIK.V_SYSCODE_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.V_SYSCODE_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.V_SYSCODE 
FOR EACH ROW
BEGIN
  SELECT V_SYSCODE_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
